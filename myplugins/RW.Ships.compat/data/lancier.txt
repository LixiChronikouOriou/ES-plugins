mission "Lancier Raid on Merchants (North)"
	name "Large merchant convoy near <system>"
	repeat
	job
	description "A large merchant fleet has been spotted near the <system> system. Join in destroying them and return to <planet> by <day> for a cut of the loot (<payment>)."
	to offer
		random < 30
		"combat rating" > 400
		"reputation: Pirate" > 0
	source
		government "Pirate"
		attributes "north pirate"
	npc
		government "Pirate"
		personality plunders harvests escort
		fleet "Small Northern Pirates"
		fleet "pirate raid"
		fleet "pirate raid (lancier)"
	npc kill
		government "Merchant"
		personality heroic staying target
		system
			distance 1 3
			neighbor
				not government "Pirate"
		fleet "Small Northern Merchants" 3
		fleet "Large Northern Merchants" 3
		dialog "The merchants have been eliminated. Return to <destination> for your cut of the loot."
	on complete
		payment 400000
		dialog "After counting up the loot from the destroyed merchant fleet, you are given your cut. <payment>"
